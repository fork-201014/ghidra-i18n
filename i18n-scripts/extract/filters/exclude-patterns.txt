# 排除模式正则列表（每行一个正则）
# 匹配以下任一模式的字符串将被排除出翻译候选

# 空字符串/空白
^\s*$

# 纯标点符号
^[\[\]{}()/:,;.\-_+=*&^%$#@!~`<>?\\|\s'\"]+$

# 全大写常量
^[A-Z_][A-Z0-9_]*$

# URL
^https?:
^file:
^ftp:

# 以数字开头
^\d

# Java package 路径
^[a-z]+(\.[a-z]+){2,}$

# 纯数字
^\d+$

# Gradle 坐标/命令行参数
^--[a-z]
^-[a-z]$

# API Key 模式
^sk-
^ghp_
^ghs_

# 文件扩展名
^\.\w+$

# MIME 类型
^\w+/\w+

# 编程语言关键字/字面量
^(true|false|null)$
